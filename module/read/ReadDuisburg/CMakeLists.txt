set(HEADERS ReadDuisburg.h)
set(SOURCES ReadDuisburg.cpp)

vistle_find_package(NETCDF)
if(NOT NETCDF_FOUND)
    return()
endif()

add_module(ReadDuisburg "Read Duisburg files" ${SOURCES} ${HEADERS})
target_compile_definitions(ReadDuisburg PRIVATE USE_NETCDF)
if(NETCDF_PARALLEL)
    target_compile_definitions(ReadDuisburg PRIVATE NETCDF_PARALLEL)
endif()
target_link_libraries(ReadDuisburg ${NETCDF_LIBRARIES} vistle_netcdf)
target_include_directories(ReadDuisburg SYSTEM PRIVATE ${NETCDF_INCLUDE_DIRS})
